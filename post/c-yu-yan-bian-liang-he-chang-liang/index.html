C语言-变量和常量

发布于 2021-09-22 | 分类于 C语言 | 2分钟 | 298字数 | 浏览量： 
变量
int b = 2021;//全局变量，在主函数外面的变量叫全局变量。
int main()
{	
printf("hehe\n");
{
	int a = 10;// 局部变量，生命周期在括号内，出括号生命结束，全局变量和局部变量冲突的情况下，局部优先。
	printf("1__%d\n", b);
	printf("a = %d\n", a);
}

printf("2__%c\n", b);
return 0;

}

常量
#define max 1000//  标识符常量
3.14；浮点型常量
10；整数常量
‘a’；字符常量
“abcd”字符串常量

int main()
{
const int num = 10;当变量被const修饰后就不能被改变
//num = 20;
printf("%d\n", num);
return 0;

}
——————————————————————————————
enum sex //枚举常量,可以被一 一列举的常量，enum 枚举关键字
{
//  这3个值就是枚举类型的变量的未来可能取值
MALE ,
//MALE = 3 ,  //赋初始值，结果就会变成3，4，5
FAMALE ,
aaaa
};
int main()
{
enum sex s = MALE;//用enum sex这个枚举类型定义一个变量s那么这个s的取值只能是上面定                                         义的3个里面的一个
	printf("%d\n", MALE);结果为0           //枚举从0开始
	printf("%d\n", FAMALE);结果为1       // 
	printf("%d\n", aaaa);结果为2             //
	return 0;

}
